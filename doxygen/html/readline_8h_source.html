164 static inline int readline_putchar(struct readline *rl, char c)
165 {
166  int ret;
167  int retcode;
168 
169  switch (rl->state)
170  {
171  case READLINE_STATE_NORMAL:
172  switch (c)
173  {
174  case '\r':
175  case '\n':
176  // TODO: Возможно тут некорректно отрабатывается комбинация rnrnrnrn
177  if ((rl->last == '\n' || rl->last == '\r') && rl->last != c)
178  {
179  rl->last = 0;
180  retcode = READLINE_NOTHING;
181  }
182  else
183  {
184  if (rl->history_space && rl->line.len &&
185  readline_is_not_same_as_last(rl))
186  {
187  // Если есть буффер истории и введенная строка не нулевая и
188  // отличается от последней, сохраняем строку в историю.
189  readline_push_current_line_to_history(rl);
190  }
191  rl->curhist = 0;
192 
193  retcode = READLINE_NEWLINE;
194  }
195  break;
196 
197  case ASCII_BS:
198  ret = sline_backspace(&rl->line, 1);
199  retcode = ret ? READLINE_BACKSPACE : READLINE_NOTHING;
200  break;
201 
202  case ASCII_ESC:
203  rl->state = READLINE_STATE_ESCSEQ;
204  retcode = READLINE_NOTHING;
205  break;
206 
207  default:
208  sline_putchar(&rl->line, c);
209  retcode = READLINE_ECHOCHAR;
210  break;
211  }
212  break;
213 
214  case READLINE_STATE_ESCSEQ:
215  switch (c)
216  {
217  case 0x5B:
218  rl->state = READLINE_STATE_ESCSEQ_MOVE;
219  break;
220 
221  default:
222  dprln("?warning in readline?");
223  rl->state = READLINE_STATE_NORMAL;
224  break;
225  }
226  retcode = READLINE_NOTHING;
227  break;
228 
229  case READLINE_STATE_ESCSEQ_MOVE:
230  retcode = READLINE_NOTHING;
231  switch (c)
232  {
233  case 0x41: // up
234  if (readline_history_up(rl))
235  retcode = READLINE_UPDATELINE;
236  break;
237  case 0x42: // down
238  if (readline_history_down(rl))
239  retcode = READLINE_UPDATELINE;
240  break;
241  case 0x43: // right
242  ret = sline_right(&rl->line);
243  if (ret)
244  retcode = READLINE_RIGHT;
245  break;
246  case 0x44: // left
247  ret = sline_left(&rl->line);
248  if (ret)
249  retcode = READLINE_LEFT;
250  break;
251  case 0x33:
252  ret = sline_delete(&rl->line, 1);
253  if (ret)
254  retcode = READLINE_DELETE;
255  rl->state = READLINE_STATE_ESCSEQ_MOVE_WAIT_7E;
256  rl->last = c;
257  return retcode;
258  default:
259  dpr("unr esc: hex:");
260  dprhex(c);
261  dpr(" dec:");
262  dprln((int)c);
263  }
264  rl->state = READLINE_STATE_NORMAL;
265  break;
266 
267  case READLINE_STATE_ESCSEQ_MOVE_WAIT_7E:
268  if (c != '\x7E')
269  dprln("?esc?");
270  rl->state = READLINE_STATE_NORMAL;
271  retcode = READLINE_NOTHING;
272  break;
273 
274  default:
275  BUG();
276  }
277 
278  rl->last = c;
279  return retcode;
280 }
